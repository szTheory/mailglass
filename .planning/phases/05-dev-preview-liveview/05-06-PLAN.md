          scenario = socket.assigns.current_scenario
          assigns_map = socket.assigns.current_assigns

          case build_and_render(mod, scenario, assigns_map) do
            {:ok, rendered} ->
              email = rendered.swoosh_email

              socket

              |> assign(:text_body, email.text_body || "")
