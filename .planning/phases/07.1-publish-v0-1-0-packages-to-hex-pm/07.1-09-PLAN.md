    **Cron-guard job (D-19 7-day-window guard):**

    All subsequent jobs `needs:` this job. The guard short-circuits scheduled runs if the latest release is older than 7 days, so the cron stops paying CI cost after the publish risk window closes:

    ```yaml
    jobs:
      cron-guard:
        name: Cron 7-day window guard
        runs-on: ubuntu-latest
        outputs:
          should_run: ${{ steps.guard.outputs.should_run }}
          version: ${{ steps.guard.outputs.version }}
        steps:
          - name: Determine if smoke should run
            id: guard
            uses: actions/github-script@60a0d83039c74a4aee543508d2ffcb1c3799cdea
            with:
              script: |
                // workflow_dispatch / workflow_run always run
                if (context.eventName !== 'schedule') {
                  const inputVersion = context.payload?.inputs?.version || "";
                  core.setOutput("should_run", "true");
                  core.setOutput("version", inputVersion);
                  core.info(`Smoke triggered by ${context.eventName}; should_run=true; version='${inputVersion}'`);
                  return;
                }

                // schedule: query latest release; skip if > 7 days old
                const owner = context.repo.owner;
                const repo = context.repo.repo;
                const {data: releases} = await github.rest.repos.listReleases({
                  owner,
                  repo,
                  per_page: 5
                });
                if (releases.length === 0) {
                  core.setOutput("should_run", "false");
                  core.info("No releases found; skipping scheduled smoke.");
                  return;
                }
                const latest = releases[0];
                const releasedAt = new Date(latest.published_at);
                const ageDays = (Date.now() - releasedAt.getTime()) / (1000 * 60 * 60 * 24);
                if (ageDays > 7) {
                  core.setOutput("should_run", "false");
                  core.info(`Latest release ${latest.tag_name} is ${ageDays.toFixed(1)} days old; skipping scheduled smoke.`);
                  return;
                }
                // Extract bare version from tag (e.g. "mailglass-sibling-group-v0.1.0" → "0.1.0")
                const tagMatch = latest.tag_name.match(/v(\d+\.\d+\.\d+(?:-[\w.+-]+)?)/);
                const version = tagMatch ? tagMatch[1] : "";
                core.setOutput("should_run", "true");
                core.setOutput("version", version);
                core.info(`Latest release ${latest.tag_name} is ${ageDays.toFixed(1)} days old; running smoke for v${version}.`);
    ```
