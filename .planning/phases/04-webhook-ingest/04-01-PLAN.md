```elixir
def generate_sendgrid_keypair do
  {pub_key, priv_key} = :crypto.generate_key(:ecdh, :secp256r1)
  spki_der = :public_key.der_encode(:SubjectPublicKeyInfo, build_spki(pub_key))
  {Base.encode64(spki_der), priv_key}
end

def sign_sendgrid_payload(timestamp, raw_body, priv_key) do
  payload = timestamp <> raw_body
  sig = :public_key.sign(payload, :sha256, {:ECPrivateKey, priv_key, build_ec_params()})
  Base.encode64(sig)
end
```

    ```elixir
        spki_der = :public_key.der_encode(:SubjectPublicKeyInfo, spki)
        {Base.encode64(spki_der), priv}
      end

      @doc "Sign timestamp+raw_body with the private key; returns base64-encoded ECDSA signature."
      @spec sign_sendgrid_payload(String.t(), binary(), term()) :: String.t()
      def sign_sendgrid_payload(timestamp, raw_body, priv_key) when is_binary(timestamp) and is_binary(raw_body) do
        payload = timestamp <> raw_body
        ec_key = {:ECPrivateKey, 1, priv_key, build_ec_params(), nil}
        sig = :public_key.sign(payload, :sha256, ec_key)
        Base.encode64(sig)
      end

      # ---- Postmark Basic Auth ----

      @doc "Build a Basic Auth header value for Postmark fixtures (matches the WebhookCase test creds)."
      @spec postmark_basic_auth_header(String.t(), String.t()) :: {String.t(), String.t()}
      def postmark_basic_auth_header(user, pass) when is_binary(user) and is_binary(pass) do
        creds = Base.encode64("#{user}:#{pass}")
        {"authorization", "Basic " <> creds}
      end

      # ---- Fixture loading ----

      @doc "Load a Postmark fixture JSON payload (returns the raw bytes)."
      @spec load_postmark_fixture(String.t()) :: binary()
      def load_postmark_fixture(name) when is_binary(name) do
        File.read!(Path.join([@fixture_root, "postmark", name <> ".json"]))
      end

      @doc "Load a SendGrid fixture JSON payload (returns the raw bytes)."
      @spec load_sendgrid_fixture(String.t()) :: binary()
      def load_sendgrid_fixture(name) when is_binary(name) do
        File.read!(Path.join([@fixture_root, "sendgrid", name <> ".json"]))
      end

      # ---- Internal helpers (DER encoding) ----

      # SubjectPublicKeyInfo per RFC 5280 §4.1 — algorithm = ecPublicKey + curve = secp256r1
      defp build_spki(pub_bits) do
        ec_oid = {1, 2, 840, 10045, 2, 1}                           # id-ecPublicKey
        secp256r1_oid = {1, 2, 840, 10045, 3, 1, 7}                 # secp256r1
        curve_params_der = :public_key.der_encode(:EcpkParameters, {:namedCurve, secp256r1_oid})
        {:SubjectPublicKeyInfo,
         {:AlgorithmIdentifier, ec_oid, curve_params_der},
         pub_bits}
      end

      defp build_ec_params do
        {:namedCurve, {1, 2, 840, 10045, 3, 1, 7}}                  # secp256r1
      end
    end
    ```
