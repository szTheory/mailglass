         @moduledoc """
         Ecto schema for the `mailglass_webhook_events` table (V02 migration,
         Plan 01). Mutable + prunable — UNLIKE `mailglass_events` which is
         append-only via SQLSTATE 45A01 trigger (CONTEXT D-15 split).

         Stores raw webhook payloads from Postmark + SendGrid for:
           - Idempotency: UNIQUE `(provider, provider_event_id)` defends
             against replay (D-15 + Pitfall MAIL-03)
           - Audit: full payload available for support/debugging

         The `:raw_payload` field is marked `redact: true` so `Inspect`
         output (test failures, IEx) does NOT leak PII bytes.
         """

         use Mailglass.Schema

         alias Mailglass.Clock

         @valid_statuses [:received, :processing, :succeeded, :failed, :dead]

         schema "mailglass_webhook_events" do
           field :tenant_id, :string
           field :provider, :string                       # 'postmark' | 'sendgrid'
           field :provider_event_id, :string
           field :event_type_raw, :string
           field :event_type_normalized, :string          # nullable until normalized
           field :status, Ecto.Enum, values: @valid_statuses
           field :raw_payload, :map, redact: true
           field :received_at, :utc_datetime_usec
           field :processed_at, :utc_datetime_usec

           timestamps(type: :utc_datetime_usec)
         end

         @doc """
         Build a changeset for inserting a webhook_event row at ingest time.
         Caller passes `:provider`, `:provider_event_id`, `:event_type_raw`,
         `:tenant_id`, `:raw_payload`. Other fields default sensibly.
         """
         @spec changeset(map()) :: Ecto.Changeset.t()
         def changeset(attrs) when is_map(attrs) do
           attrs_with_defaults =
             attrs
             |> Map.put_new(:status, :processing)
             |> Map.put_new(:received_at, Clock.utc_now())

           %__MODULE__{}
           |> Ecto.Changeset.cast(attrs_with_defaults, [
                :tenant_id, :provider, :provider_event_id, :event_type_raw,
                :event_type_normalized, :status, :raw_payload, :received_at, :processed_at

                :tenant_id, :provider, :provider_event_id, :event_type_raw,
                :status, :raw_payload, :received_at
              ])
         end

         @doc "Returns the closed set of valid `:status` atoms."
         @spec __statuses__() :: [atom()]

       end

    ```elixir
    defmodule Mailglass.Webhook.Ingest do
      @moduledoc """
      Webhook ingest — the single Ecto.Multi that HOOK-06 reduces to.

      ## Composition (CONTEXT D-15 amended HOOK-06)

      Inside `Mailglass.Repo.transact/1`:

        1. `SET LOCAL statement_timeout = '2s'` (D-29 — DoS bound)
        2. `SET LOCAL lock_timeout = '500ms'` (D-29)
        3. `Multi.insert(:webhook_event, ...)` with
           `on_conflict: :nothing, conflict_target: [:provider, :provider_event_id]`
           — UNIQUE collision = replay no-op (returns the row but doesn't insert)
        4. For each `%Event{}` in the input list:
             a. `Events.append_multi(:event_<i>, fn changes -> attrs end)`
                — function form; `attrs.delivery_id` resolved by lookup
             b. If matched delivery: `Multi.update({:projector, i}, ...)` calling
                `Projector.update_projections/2`
             c. If orphan: SKIP projector step; event row inherits
                `delivery_id: nil + needs_reconciliation: true`
        5. `Multi.update_all(:flip_status, ...)` setting webhook_event.status = :succeeded

      ## Replay semantics

      UNIQUE collision on `(provider, provider_event_id)` is a structural
      no-op — `on_conflict: :nothing` returns the original row without
      INSERT. The Ingest module detects this via `id == nil` on the
      returned changeset (Postgres returns the conflict-target row, not
      the original on `:nothing`) and surfaces `%{duplicate: true}` in
      the changes map. Plug returns 200 either way.

      ## Orphan path

      A normalized event whose `provider_event_id` doesn't match any
      `mailglass_deliveries.provider_message_id` is an "orphan" — the
      webhook arrived before the Delivery row committed (race window).
      Per CONTEXT D-15 + Pitfall 4: the event row inserts with
      `delivery_id: nil + needs_reconciliation: true` AND the projector
      step is SKIPPED for that event (`Projector.update_projections/2`
      pattern-matches `%Delivery{}` and would FunctionClauseError on nil).

      Plan 07's `Mailglass.Webhook.Reconciler` Oban cron sweeps these
      orphans and appends a `:reconciled` event when the matching
      Delivery later commits (D-18 — append, never UPDATE).

      ## Output shape

      ```
      {:ok, %{
        webhook_event: %WebhookEvent{},
        duplicate: true | false,
        events_with_deliveries: [{event, delivery}, ...],   # for Plug post-commit broadcast
        orphan_event_count: non_neg_integer()
      }}
      ```

      Or `{:error, reason}` if the transact/1 raises (SQLSTATE 45A01 from
      append-only ledger violation, etc.).
      """

      import Ecto.Query

      alias Ecto.Multi
      alias Mailglass.{Clock, Events, IdempotencyKey, Repo}
      alias Mailglass.Events.Event
      alias Mailglass.Outbound.{Delivery, Projector}
      alias Mailglass.Tenancy
      alias Mailglass.Webhook.WebhookEvent

      @doc """
      Ingest a verified webhook into the persistence layer.

      ## Args

        * `provider` — `:postmark | :sendgrid`
        * `raw_body` — verified raw bytes (Plug ensures verify!/3 returned :ok)
        * `events` — list of `%Mailglass.Events.Event{}` from `Provider.normalize/2`

      ## Returns

      See module doc for the full output shape. Caller (Plug) iterates
      `:events_with_deliveries` to call `Projector.broadcast_delivery_updated/3`
      AFTER this function returns `{:ok, _}` (Phase 3 D-04 — broadcast post-commit).
      """
      @spec ingest_multi(atom(), binary(), [Event.t()]) ::
              {:ok, map()} | {:error, term()}
      def ingest_multi(provider, raw_body, events)
          when provider in [:postmark, :sendgrid] and is_binary(raw_body) and is_list(events) do
        tenant_id =
          Tenancy.current() ||
            raise Mailglass.TenancyError.new(:unstamped,
                    context: %{operation: :webhook_ingest})

        # Per revision B2 + CONTEXT D-11: guard against :async at v0.1
        # (the schema entry is reserved but unimplemented; raises until v0.5 ships
        # the DLQ admin that justifies async ingest).
        case Mailglass.Config.webhook_ingest_mode() do
          :sync -> :ok
          :async -> raise "webhook_ingest_mode: :async is reserved for v0.5 (CONTEXT D-11); v0.1 supports :sync only"
        end

        Repo.transact(fn ->
          # CONTEXT D-29: SET LOCAL inside the transaction (Pitfall 6)
          _ = Repo.query!("SET LOCAL statement_timeout = '2s'", [])
          _ = Repo.query!("SET LOCAL lock_timeout = '500ms'", [])

          multi = build_multi(provider, raw_body, events, tenant_id)

          case Repo.multi(multi) do
            {:ok, changes} ->
              {:ok, finalize_changes(changes, events)}

            {:error, _step, reason, _changes} ->
              {:error, reason}
          end
        end)
      end

      # ---- Multi composition ----

      defp build_multi(provider, raw_body, events, tenant_id) do
        provider_event_id = derive_webhook_provider_event_id(provider, raw_body, events)

        # Step 0 (per revision B5): deterministic duplicate signal via pre-insert lookup.
        # We do this INSIDE the same transaction (Multi.run) so the read sees the same
        # snapshot as the upcoming insert. If the row already exists, on_conflict: :nothing
        # below is a no-op AND we surface the duplicate flag for the Plug's response handling.
        duplicate_check_step =
          Multi.run(Multi.new(), :duplicate_check, fn _repo, _changes ->
            exists? =
              Repo.one(
                from(w in WebhookEvent,
                  where: w.provider == ^Atom.to_string(provider) and w.provider_event_id == ^provider_event_id,
                  select: true,
                  limit: 1
                )
              ) == true

            {:ok, exists?}
          end)

        # Step 1: insert webhook_event with UNIQUE replay protection
        webhook_event_attrs = %{
          tenant_id: tenant_id,
          provider: Atom.to_string(provider),
          provider_event_id: provider_event_id,
          event_type_raw: derive_event_type_raw(events),
          event_type_normalized: derive_event_type_normalized(events),
          status: :processing,
          raw_payload: parse_raw_payload(raw_body),
          received_at: Clock.utc_now()
        }

        duplicate_check_step
        |> Multi.insert(:webhook_event, WebhookEvent.changeset(webhook_event_attrs),
             on_conflict: :nothing,
             conflict_target: [:provider, :provider_event_id],
             returning: true)
        |> append_events_for_each(events, provider, tenant_id)
        |> update_projections_for_each(events)
        |> Multi.update_all(:flip_status, &flip_status_query(&1, provider), set: [
             status: :succeeded,
             processed_at: Clock.utc_now()
           ])
      end

      # Step 2: for each Event in the list, run Events.append_multi with a function
      # that resolves delivery_id from the prior changes (the webhook_event row id
      # is available, but we look up Delivery by provider_message_id which is the
      # Event's provider_event_id — these are the SAME value for SendGrid + Postmark
      # because they identify the message, not the event).
      defp append_events_for_each(multi, events, provider, tenant_id) do
        events
        |> Enum.with_index()
        |> Enum.reduce(multi, fn {event, idx}, acc ->
          Events.append_multi(acc, {:event, idx}, fn _changes ->
            delivery_id = resolve_delivery_id(provider, event)

            %{
              type: event.type,
              provider: Atom.to_string(provider),
              tenant_id: tenant_id,
              delivery_id: delivery_id,
              needs_reconciliation: is_nil(delivery_id),
              idempotency_key:
                IdempotencyKey.for_webhook_event(provider, event.provider_event_id, idx),
              metadata: event.metadata || %{},
              reject_reason: event.reject_reason,
              occurred_at: Clock.utc_now()
            }
          end)
        end)
      end

      # Step 3: project for each event with matched delivery (Pitfall 4 — skip orphans).
      # Per revision W4 — refactored to call Multi.update directly on the outer Multi
      # rather than nesting Repo.multi(Multi.new() |> Multi.update(...)) inside Multi.run.
      # The previous nested-Multi pattern was an anti-pattern: the outer transaction can't
      # rollback the inner sub-transaction's writes if a later step fails. Direct
      # Multi.update + Multi.run categorization keeps everything in one transaction.
      defp update_projections_for_each(multi, events) do
        events
        |> Enum.with_index()
        |> Enum.reduce(multi, fn {_event, idx}, acc ->
          # First, categorize: lookup the inserted event + its delivery (if any).
          # Returns one of: :no_event_row, :orphan_skipped, {:matched, delivery, inserted_event}
          acc =
            Multi.run(acc, {:projector_categorize, idx}, fn _repo, changes ->
              inserted_event = Map.get(changes, {:event, idx})

              cond do
                is_nil(inserted_event) ->
                  {:ok, :no_event_row}

                is_nil(inserted_event.delivery_id) ->
                  {:ok, :orphan_skipped}

                true ->
                  case Repo.get(Delivery, inserted_event.delivery_id) do
                    nil -> {:ok, :orphan_skipped}
                    delivery -> {:ok, {:matched, delivery, inserted_event}}
                  end
              end
            end)

          # Then, conditionally update the projection in the outer Multi (no nesting).
          Multi.run(acc, {:projector_apply, idx}, fn repo, changes ->
            case Map.get(changes, {:projector_categorize, idx}) do
              {:matched, delivery, inserted_event} ->
                changeset = Projector.update_projections(delivery, inserted_event)
                case repo.update(changeset) do
                  {:ok, _projected} -> {:ok, {delivery, inserted_event}}
                  {:error, reason} -> {:error, reason}
                end

              other ->
                # :no_event_row or :orphan_skipped — pass through
                {:ok, other}
            end
          end)
        end)
      end

      # Step 4: flip status — uses an update_all referencing the inserted webhook_event id
      defp flip_status_query(changes, provider) do
        webhook_event = Map.fetch!(changes, :webhook_event)

        from(w in WebhookEvent,
          where: w.id == ^webhook_event.id and w.provider == ^Atom.to_string(provider),
          update: [set: [status: :succeeded, processed_at: ^Clock.utc_now()]]
        )
      end

      # ---- Helpers ----

      # Per revision B6 — SendGrid batch idempotency requires the raw_body SHA-256 hash
      # discriminator. Plain "first event id" breaks: SendGrid retries the SAME batch
      # (identical bytes) but the V02 UNIQUE(provider, provider_event_id) constraint
      # would NOT collide if a SECOND batch with a DIFFERENT first event happens to
      # share that batch's first-event-id (e.g., batch A: [evt_X, evt_Y]; batch B:
      # [evt_X, evt_Z] — both compute discriminator "evt_X", masking the second
      # batch's events as a duplicate). Hash-of-raw-body fixes this.
      defp derive_webhook_provider_event_id(:sendgrid, raw_body, _events) when is_binary(raw_body) do
        :crypto.hash(:sha256, raw_body)
        |> Base.encode16(case: :lower)
        |> binary_part(0, 32)
      end

      # Postmark sends one event per webhook; provider_event_id is canonical.
      defp derive_webhook_provider_event_id(:postmark, _raw_body, [first | _]) do
        first.provider_event_id
      end

      defp derive_webhook_provider_event_id(_provider, _raw_body, []), do: ""

      defp derive_event_type_raw(events) do
        events
        |> Enum.map(fn e -> e.metadata[:event] || e.metadata[:record_type] || "unknown" end)
        |> Enum.uniq()
        |> Enum.join(",")
      end

      defp derive_event_type_normalized(events) do
        events
        |> Enum.map(& Atom.to_string(&1.type))
        |> Enum.uniq()
        |> Enum.join(",")
      end

      defp parse_raw_payload(raw_body) when is_binary(raw_body) do
        case Jason.decode(raw_body) do
          {:ok, payload} -> payload
          _ -> %{"_raw" => raw_body}
        end
      end

      # Lookup Delivery by (provider, provider_message_id == event.provider_event_id)
      # NOTE: For some providers, provider_event_id is per-EVENT (e.g., sg_event_id),
      # not per-MESSAGE. The lookup may need to use a different field — check the
      # Event struct for a separate provider_message_id field that the provider
      # normalizers populate from sg_message_id / Postmark MessageID.
      # Per revision W9 — read STRING keys from metadata (Plans 02 + 03 normalize/2
      # standardized on string keys for JSONB roundtrip safety; this code reads the
      # same string keys + retains atom-key fallback for defensive compatibility with
      # any pre-W9 callers that might exist temporarily).
      defp resolve_delivery_id(provider, event) do
        meta = event.metadata || %{}
        message_id =
          meta["sg_message_id"] || meta["message_id"] ||
            Map.get(meta, :sg_message_id) || Map.get(meta, :message_id)

        cond do
          is_nil(message_id) ->
            nil

          true ->
            from(d in Delivery,
              where: d.provider == ^Atom.to_string(provider) and d.provider_message_id == ^message_id,
              select: d.id,
              limit: 1
            )
            |> Repo.one()
        end
      end

      # Build the final result map for the Plug
      defp finalize_changes(changes, events) do
        webhook_event = Map.get(changes, :webhook_event)

        # Detect duplicate via deterministic pre-insert lookup (per revision B5).
        # The previous heuristic `is_nil(webhook_event.id)` is structurally broken —
        # `on_conflict: :nothing, returning: true` returns the conflict-target row WITH
        # its existing id, so the id is NEVER nil after insert/conflict. The deterministic
        # signal is set by build_multi/4 BEFORE the Multi runs: a Repo.exists?/1 lookup
        # against (provider, provider_event_id) — see :duplicate_check Multi step added in
        # build_multi/4 below. The flag is propagated through the Multi changes map.
        duplicate? = Map.get(changes, :duplicate_check, false)

        events_with_deliveries =
          events
          |> Enum.with_index()
          |> Enum.flat_map(fn {_event, idx} ->
            case Map.get(changes, {:projector_apply, idx}) do
              {delivery, inserted_event} -> [{inserted_event, delivery}]
              _ -> []
            end
          end)

        orphan_event_count =
          events
          |> Enum.with_index()
          |> Enum.count(fn {_event, idx} ->
            Map.get(changes, {:projector_apply, idx}) == :orphan_skipped
          end)

        %{
          webhook_event: webhook_event,
          duplicate: duplicate?,
          events_with_deliveries: events_with_deliveries,
          orphan_event_count: orphan_event_count
        }
      end
    end
    ```
