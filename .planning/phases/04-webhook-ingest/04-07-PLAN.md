    1. **lib/mailglass/webhook/reconciler.ex** — NEW Oban-conditional worker. Module skeleton (entire file wrapped in `if Code.ensure_loaded?(Oban.Worker)`):
       ```elixir
       defmodule Mailglass.Webhook.Reconciler do
         @moduledoc """
         Oban cron worker that closes the orphan-webhook race window
         (CONTEXT D-17, D-18).

         An orphan webhook event is one inserted by `Mailglass.Webhook.Ingest`
         with `delivery_id: nil + needs_reconciliation: true` because the
         matching `mailglass_deliveries` row had not yet committed when the
         webhook arrived (a real race in production for low-latency
         providers).

         This worker runs every 5 minutes (`*/5 * * * *` cron — adopters
         wire this in their Oban config; see `guides/webhooks.md`). For each
         orphan older than 60 seconds (grace window for late commits):

           1. `Mailglass.Events.Reconciler.find_orphans/1` returns the candidate batch
           2. `Mailglass.Events.Reconciler.attempt_link/2` looks up the Delivery
           3. On match: append a NEW `:reconciled` event (D-18 — append, never
              UPDATE; preserves the SQLSTATE 45A01 append-only invariant) +
              call `Projector.update_projections/2` for the matched Delivery
           4. On no-match: leave the orphan row untouched; next tick retries

         After 7 days, `find_orphans/1` filters the row out (admin LiveView
         shows it as "older than 7 days — unlikely to reconcile" per D-19).

         ## Optional-dep gating

         The entire module is conditionally compiled behind
         `if Code.ensure_loaded?(Oban.Worker) do ... end`. When Oban is
         absent, the module simply does not exist; `Mailglass.Application`
         emits a one-time `Logger.warning` (D-20) directing operators to
         either run `mix mailglass.reconcile` or add Oban to their deps.
         """

         if Code.ensure_loaded?(Oban.Worker) do
           use Oban.Worker, queue: :mailglass_reconcile, unique: [period: 60]

           require Logger

           alias Ecto.Multi
           alias Mailglass.{Clock, Events, Repo}
           alias Mailglass.Events.Reconciler, as: EventsReconciler
           alias Mailglass.Outbound.Projector

           @grace_seconds 60
           @max_age_minutes 7 * 24 * 60
           @batch_limit 1000

           @impl Oban.Worker
           def perform(%Oban.Job{args: args} = job) do
             tenant_id = Map.get(args, "tenant_id")
             limit = Map.get(args, "limit", @batch_limit)

             # Phase 2 shipped Oban tenancy middleware — wrap perform body if available
             # If TenancyMiddleware exposes wrap_perform/2, use it; else just run inline.
             reconcile(tenant_id, limit)
           end

           @doc """
           Run the reconciliation sweep for the given tenant (or all tenants if nil).
           Returns `{:ok, %{scanned: n, linked: m}}`.
           """
           @spec reconcile(String.t() | nil, pos_integer()) :: {:ok, map()}
           def reconcile(tenant_id, limit \\ @batch_limit) do
             :telemetry.span(
               [:mailglass, :webhook, :reconcile],
               %{tenant_id: tenant_id, status: :pending},
               fn ->
                 orphans =
                   EventsReconciler.find_orphans(
                     tenant_id: tenant_id,
                     limit: limit,
                     min_age_seconds: @grace_seconds,
                     max_age_minutes: @max_age_minutes
                   )

                 linked =
                   Enum.reduce(orphans, 0, fn orphan, acc ->
                     case attempt_reconcile(orphan) do
                       {:ok, _} -> acc + 1
                       {:error, :no_match} -> acc
                       {:error, reason} ->
                         Logger.warning(
                           "[mailglass] Reconcile attempt failed for orphan #{orphan.id}: #{inspect(reason)}"
                         )
                         acc
                     end
                   end)

                 result = {:ok, %{scanned: length(orphans), linked: linked}}

                 meta = %{
                   tenant_id: tenant_id,
                   scanned_count: length(orphans),
                   linked_count: linked,
                   remaining_orphan_count: max(length(orphans) - linked, 0),
                   status: :ok
                 }

                 {result, meta}
               end
             )
           end

           # CONTEXT D-18: APPEND a :reconciled event; do NOT UPDATE the orphan row
           # Per revision W6 — flat Multi composition (no nested Repo.multi inside
           # Multi.run). The previous nested-Multi pattern had the same anti-pattern
           # as Plan 06's update_projections_for_each (now also fixed): the outer
           # transaction can't rollback the inner sub-transaction's writes if a
           # later step fails. Direct Multi.run + Multi.update keeps everything
           # atomic in one transaction.
           defp attempt_reconcile(orphan) do
             case EventsReconciler.attempt_link(orphan) do
               {:ok, {delivery, ^orphan}} ->
                 reconciled_attrs = %{
                   type: :reconciled,
                   delivery_id: delivery.id,
                   tenant_id: orphan.tenant_id,
                   metadata: %{
                     "reconciled_from_event_id" => orphan.id,
                     "reconciled_provider" => orphan.provider,
                     "reconciled_provider_event_id" => extract_provider_event_id(orphan)
                   },
                   idempotency_key: "reconciled:" <> to_string(orphan.id),
                   occurred_at: Clock.utc_now()
                 }

                 # Build flat Multi: append event → derive projection changeset → update.
                 # No nested Repo.multi inside Multi.run.
                 multi =
                   Multi.new()
                   |> Events.append_multi(:reconciled_event, reconciled_attrs)
                   |> Multi.run(:projection_changeset, fn _repo, %{reconciled_event: e} ->
                     {:ok, Projector.update_projections(delivery, e)}
                   end)
                   |> Multi.update(:projection, fn %{projection_changeset: cs} -> cs end)

                 case Repo.transact(fn -> Repo.multi(multi) end) do
                   {:ok, {:ok, changes}} ->
                     # Post-commit broadcast (Phase 3 D-04 invariant)
                     reconciled_event = changes[:reconciled_event]
                     if reconciled_event do
                       Projector.broadcast_delivery_updated(delivery, :reconciled, %{
                         event_id: reconciled_event.id,
                         reconciled_from_event_id: orphan.id
                       })
                     end
                     {:ok, changes}

                   {:ok, {:error, _step, reason, _}} ->
                     {:error, reason}

                   {:error, reason} ->
                     {:error, reason}
                 end

               {:error, :delivery_not_found} ->
                 {:error, :no_match}

               {:error, reason} ->
                 {:error, reason}
             end
           end

           defp extract_provider_event_id(orphan) do
             case orphan.metadata do
               %{"provider_event_id" => id} -> id
               %{provider_event_id: id} -> id
               _ -> nil
             end
           end
         end
       end
       ```

       defmodule Mailglass.Webhook.Pruner do
         @moduledoc """
         Oban cron worker that prunes `mailglass_webhook_events` rows by
         status + age (CONTEXT D-16).

         Three retention knobs (per `Mailglass.Config :webhook_retention`):

           * `:succeeded_days` (default 14) — prune succeeded after N days
           * `:dead_days` (default 90) — prune dead-after-retries after N days
           * `:failed_days` (default `:infinity`) — investigatable; never prune by default

         Set any knob to `:infinity` to disable that prune class.

         ## Optional-dep gating

         Conditionally compiled behind `if Code.ensure_loaded?(Oban.Worker)`.
         Oban-absent: adopters run `mix mailglass.webhooks.prune` from their
         own cron infrastructure.

         ## GDPR erasure

         Targeted DELETE on `mailglass_webhook_events.raw_payload->>'to' = ?`
         is the GDPR path (D-15) — handled by adopter ad-hoc, NOT this Pruner.
         """

         if Code.ensure_loaded?(Oban.Worker) do
           use Oban.Worker, queue: :mailglass_maintenance

           import Ecto.Query

           alias Mailglass.{Clock, Repo}
           alias Mailglass.Webhook.WebhookEvent

           @impl Oban.Worker
           def perform(_job) do
             {:ok, %{succeeded: succeeded, dead: dead}} = prune()
             :ok = emit_telemetry(succeeded, dead)
             :ok
           end

           @doc "Run the prune sweep. Returns `{:ok, %{succeeded: n, dead: m}}`."
           @spec prune() :: {:ok, %{succeeded: non_neg_integer(), dead: non_neg_integer()}}
           def prune do
             retention = Application.get_env(:mailglass, :webhook_retention, [])
             succeeded_days = retention[:succeeded_days] || 14
             dead_days = retention[:dead_days] || 90

             {:ok, succeeded_count} = prune_status(:succeeded, succeeded_days)
             {:ok, dead_count} = prune_status(:dead, dead_days)

             {:ok, %{succeeded: succeeded_count, dead: dead_count}}
           end

           # :infinity bypass — return {:ok, 0} without touching the DB
           defp prune_status(_status, :infinity), do: {:ok, 0}

           defp prune_status(status, days) when is_atom(status) and is_integer(days) and days > 0 do
             cutoff = DateTime.add(Clock.utc_now(), -days * 86_400, :second)

             {count, _} =
               from(w in WebhookEvent,
                 where: w.status == ^status and w.inserted_at < ^cutoff
               )
               |> Repo.repo().delete_all()

             {:ok, count}
           end

           defp emit_telemetry(succeeded_deleted, dead_deleted) do
             :telemetry.execute(
               [:mailglass, :webhook, :prune, :stop],
               %{succeeded_deleted: succeeded_deleted, dead_deleted: dead_deleted},
               %{status: :ok}
             )
             :ok
           end
         end
       end
