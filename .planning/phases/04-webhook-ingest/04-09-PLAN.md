    1. **test/mailglass/properties/webhook_idempotency_convergence_test.exs** — HOOK-07 1000-replay convergence:
       ```elixir
       defmodule Mailglass.Properties.WebhookIdempotencyConvergenceTest do
         @moduledoc """
         HOOK-07: any sequence of (webhook_event, replay_count 1..10)
         applied N times converges to the same state as applying each
         event once.

         Generates 1000 scenarios (CONTEXT D-27 #1). The unit-of-work is
         `Mailglass.Webhook.Ingest.ingest_multi/3` (Plan 06), driven via
         the Postmark provider for simplicity.
         """

         use ExUnit.Case, async: false
         use ExUnitProperties

         import Ecto.Query

         alias Ecto.Adapters.SQL.Sandbox
         alias Mailglass.{Repo, TestRepo, Tenancy}
         alias Mailglass.Events.Event
         alias Mailglass.Webhook.{Ingest, WebhookEvent}
         alias Mailglass.Events.Event, as: EventStruct

         @moduletag timeout: :infinity

         setup do
           Sandbox.mode(TestRepo, :auto)
           :ok = Tenancy.put_current("default")
           TestRepo.query!("TRUNCATE TABLE mailglass_webhook_events CASCADE", [])
           TestRepo.query!("TRUNCATE TABLE mailglass_events CASCADE", [])

           on_exit(fn ->
             TestRepo.query!("TRUNCATE TABLE mailglass_webhook_events CASCADE", [])
             TestRepo.query!("TRUNCATE TABLE mailglass_events CASCADE", [])
             Process.delete(:mailglass_tenant)
             Sandbox.mode(TestRepo, :manual)
           end)

           :ok
         end

         # Generator: synthetic Postmark event with random RecordType + MessageID
         defp event_gen do
           gen all record_type <- member_of(["Delivery", "Open", "Click", "SpamComplaint"]),
                   msg_id <- string(:alphanumeric, min_length: 8, max_length: 24),
                   event_id <- string(:alphanumeric, min_length: 8, max_length: 24) do
             type =
               case record_type do
                 "Delivery" -> :delivered
                 "Open" -> :opened
                 "Click" -> :clicked
                 "SpamComplaint" -> :complained
               end

             %EventStruct{
               type: type,
               provider: :postmark,
               provider_event_id: "#{record_type}:#{event_id}:2026-04-23T12:00:00Z",
               metadata: %{record_type: record_type, message_id: msg_id},
               reject_reason: nil
             }
           end
         end

         property "convergence: apply N times == apply once for any (event, replay_count) sequence" do
           check all(events <- list_of(event_gen(), min_length: 1, max_length: 10),
                     replay_count <- integer(1..10),
                     max_runs: 1000) do
             # Wipe between iterations
             TestRepo.query!("TRUNCATE TABLE mailglass_webhook_events CASCADE", [])
             TestRepo.query!("TRUNCATE TABLE mailglass_events CASCADE", [])

             # Apply each event `replay_count` times
             for _ <- 1..replay_count, event <- events do
               raw_body = ~s({"RecordType":"#{event.metadata.record_type}","MessageID":"#{event.metadata.message_id}"})
               {:ok, _} = Ingest.ingest_multi(:postmark, raw_body, [event])
             end

             # Snapshot: webhook_events count + events count + per-(provider, provider_event_id) tally
             webhook_event_count = Repo.aggregate(WebhookEvent, :count)
             event_count = Repo.aggregate(Event, :count)

             # Convergence: replay_count is irrelevant — count of webhook_events == unique provider_event_ids in input
             unique_provider_event_ids = events |> Enum.map(& &1.provider_event_id) |> Enum.uniq() |> length()
             assert webhook_event_count == unique_provider_event_ids

             # Each event row gets one entry per unique webhook_event (the function-form append_multi
             # generates a deterministic idempotency_key per (provider, provider_event_id, idx))
             assert event_count == unique_provider_event_ids
           end
         end
       end
       ```

       Notes:
       - The convergence assertion is structural: regardless of how many times we replay, the row count converges to the count of unique `(provider, provider_event_id)` tuples — UNIQUE enforces this at the DB level.
       - Per-event idempotency_key UNIQUE index also catches duplicate event-row inserts at the partial-UNIQUE level (Phase 2 shipped).
       - `max_runs: 1000` keeps the property runtime under ~30s on a typical machine.

    2. **test/mailglass/properties/webhook_signature_failure_test.exs** — TEST-03 D-27 #2:

       ```elixir
       defmodule Mailglass.Properties.WebhookSignatureFailureTest do
         @moduledoc """
         TEST-03 D-27 #2: every webhook signature failure raises EXACTLY
         ONE of the seven `Mailglass.SignatureError.type` atoms.

         Generates random (provider, mutation) tuples and verifies:
           1. The raised exception is %SignatureError{}
           2. The :type atom is in the closed set of 7 (CONTEXT D-21)
           3. NO partial DB writes happen (event count == 0 after raise)
         """

         use Mailglass.WebhookCase, async: false
         use ExUnitProperties

         import Ecto.Query

         alias Mailglass.{Repo, SignatureError}
         alias Mailglass.Events.Event
         alias Mailglass.Webhook.WebhookEvent

         @moduletag timeout: :infinity

         @valid_atoms [:missing_header, :malformed_header, :bad_credentials,
                       :ip_disallowed, :bad_signature, :timestamp_skew, :malformed_key]

         setup do
           Repo.delete_all(WebhookEvent)
           Repo.delete_all(Event)
           on_exit(fn ->
             Repo.delete_all(WebhookEvent)
             Repo.delete_all(Event)
           end)
           :ok
         end

         # Mutation generator — produces conn-shapes that should fail signature verify
         defp mutation_gen do
           gen all variant <- member_of([:missing_auth, :bad_basic_auth, :malformed_b64,
                                          :wrong_user, :wrong_pass]) do
             variant
           end
         end

         property "every Postmark signature failure raises exactly one of 7 atoms; no partial writes" do
           check all(mutation <- mutation_gen(), max_runs: 200) do
             Repo.delete_all(WebhookEvent)
             Repo.delete_all(Event)

             body = stub_postmark_fixture("delivered")
             headers = build_postmark_headers(mutation)
             config = %{basic_auth: {"test_user", "test_pass"}, ip_allowlist: []}

             raised =
               try do
                 Mailglass.Webhook.Providers.Postmark.verify!(body, headers, config)
                 nil
               rescue
                 e in SignatureError -> e
               end

             # Some mutations may raise ConfigError (missing basic_auth) — those are
             # legitimate distinct error paths. Both Mailglass.ConfigError and
             # Mailglass.SignatureError are valid raised exceptions for this property.
             # We assert: IF SignatureError raised, the atom is in the closed set.
             if raised do
               assert raised.type in @valid_atoms,
                      "Got unexpected SignatureError atom: #{inspect(raised.type)} (must be in #{inspect(@valid_atoms)})"
               assert raised.provider == :postmark
             end

             # Critical: no partial DB writes regardless of which exception raised
             assert Repo.aggregate(WebhookEvent, :count) == 0
             assert Repo.aggregate(Event, :count) == 0
           end
         end

         defp build_postmark_headers(:missing_auth), do: []
         defp build_postmark_headers(:bad_basic_auth) do
           [{"authorization", "Bearer foo"}]
         end
         defp build_postmark_headers(:malformed_b64) do
           [{"authorization", "Basic %%%not-base64%%%"}]
         end
         defp build_postmark_headers(:wrong_user) do
           {h, v} = Mailglass.WebhookFixtures.postmark_basic_auth_header("wrong_user", "test_pass")
           [{h, v}]
         end
         defp build_postmark_headers(:wrong_pass) do
           {h, v} = Mailglass.WebhookFixtures.postmark_basic_auth_header("test_user", "wrong_pass")
           [{h, v}]
         end
       end
       ```

       ```elixir
       defmodule Mailglass.Properties.WebhookTenantResolutionTest do
         @moduledoc """
         TEST-03 D-27 #3: tenant resolution via SingleTenant +

               ctx = %{
                 provider: provider,
                 conn: nil,
                 raw_body: raw_body,
                 headers: [],
                 path_params: %{},
                 verified_payload: nil
               }
               assert {:ok, "default"} = SingleTenant.resolve_webhook_tenant(ctx)
             end
           end
         end

         describe "ResolveFromPath stamps path_params['tenant_id']" do
           property "returns {:ok, tid} when tenant_id is non-empty binary" do
             check all(tid <- string(:alphanumeric, min_length: 1, max_length: 50),
                       max_runs: 100) do
               ctx = %{
                 provider: :postmark,
                 conn: nil,
                 raw_body: "",
                 headers: [],
                 path_params: %{"tenant_id" => tid},
                 verified_payload: nil
               }
               assert {:ok, ^tid} = ResolveFromPath.resolve_webhook_tenant(ctx)
             end
           end

           property "returns {:error, :missing_path_param} when tenant_id absent or empty" do
             check all(absent_or_empty <- member_of([%{}, %{"tenant_id" => ""}, %{"other" => "val"}]),

               ctx = %{
                 provider: :postmark,
                 conn: nil,
                 raw_body: "",
                 headers: [],
                 path_params: absent_or_empty,
                 verified_payload: nil
               }
               assert {:error, :missing_path_param} = ResolveFromPath.resolve_webhook_tenant(ctx)
             end
           end
         end

         describe "Bad strategy via Tenancy dispatcher raises clean TenancyError" do
           defmodule BadTenancy do
             def scope(q, _), do: q
             def resolve_webhook_tenant(_), do: {:error, :always_broken}
           end

           property "Tenancy.resolve_webhook_tenant returns {:error, _} which the Plug rescues to %TenancyError{}" do
             check all(provider <- member_of([:postmark, :sendgrid]), max_runs: 50) do
               Application.put_env(:mailglass, :tenancy, BadTenancy)
               try do
                 ctx = %{provider: provider, conn: nil, raw_body: "", headers: [], path_params: %{}, verified_payload: nil}

                 assert {:error, :always_broken} = result
               after

               end

           end
         end
       ```

       # Mailglass Webhooks Guide

       This guide walks you through mounting Mailglass webhook ingest in
       your Phoenix app. mailglass ships verifiers for Postmark
       (Basic Auth) and SendGrid (ECDSA P-256); v0.5 will add Mailgun,

       ## 1. Install + endpoint wiring

       ### Step 1: Configure Plug.Parsers in your endpoint

       Webhook signature verification needs the raw request bytes, so you
       must configure `Plug.Parsers` with mailglass's `CachingBodyReader`:

       ```elixir
       # lib/my_app_web/endpoint.ex
       plug Plug.Parsers,
         parsers: [:json],
         pass: ["*/*"],
         json_decoder: Jason,
         body_reader: {Mailglass.Webhook.CachingBodyReader, :read_body, []},
         length: 10_000_000   # 10 MB cap accommodates SendGrid batches up to 128 events
       ```

       The 10 MB cap is critical for SendGrid — their batched event
       webhooks can carry up to 128 events per POST.

       > **Footgun:** `Plug.Parsers.MULTIPART` does NOT honor `:body_reader`
       > (Plug issue #884). If you add `:multipart` to the parsers list for
       > another route, those requests bypass mailglass's CachingBodyReader.
       > Use a separate Plug.Parsers config for multipart routes.

       ### Step 2: Mount the routes in your router

       ```elixir
       # lib/my_app_web/router.ex
       defmodule MyAppWeb.Router do
         use Phoenix.Router
         import Mailglass.Webhook.Router

         pipeline :mailglass_webhooks do
           plug :accepts, ["json"]
           # NO :browser, :fetch_session, :protect_from_forgery — webhooks have none
         end

         scope "/", MyAppWeb do
           pipe_through :mailglass_webhooks
           mailglass_webhook_routes "/webhooks"
         end
       end
       ```

       This generates two POST routes:
         * `POST /webhooks/postmark` → handled by `Mailglass.Webhook.Plug`
         * `POST /webhooks/sendgrid` → handled by `Mailglass.Webhook.Plug`

       ### Step 3: Configure provider credentials

       ```elixir
       # config/runtime.exs
       config :mailglass, :postmark,
         enabled: true,
         basic_auth: {System.fetch_env!("POSTMARK_WEBHOOK_USER"), System.fetch_env!("POSTMARK_WEBHOOK_PASS")}

       config :mailglass, :sendgrid,
         enabled: true,
         public_key: System.fetch_env!("SENDGRID_WEBHOOK_PUBLIC_KEY"),
         timestamp_tolerance_seconds: 300
       ```

       SendGrid's public key is base64-encoded **DER** (NOT PEM). Get it
       from your SendGrid Event Webhook security settings.

       ## 2. Multi-tenant patterns (D-12)

       mailglass supports three multi-tenant strategies via the
       `Mailglass.Tenancy.resolve_webhook_tenant/1` optional callback.

       ### Strategy A: Single-tenant (default — zero config)

       No setup required. All events stamp `tenant_id: "default"`.

       ### Strategy B: URL prefix (built-in sugar)

       ```elixir
       config :mailglass, tenancy: Mailglass.Tenancy.ResolveFromPath
       ```

       Mount your routes with a `:tenant_id` path parameter:

       ```elixir
       scope "/tenants/:tenant_id" do
         pipe_through :mailglass_webhooks
         mailglass_webhook_routes "/webhooks"
       end
       ```

       Postmark POSTs to `/tenants/customer_a/webhooks/postmark` will stamp
       `tenant_id: "customer_a"`.

       > Note: `Mailglass.Tenancy.ResolveFromPath` does NOT implement
       > `scope/2` (it's sugar for the webhook callback only). Pair it with
       > a Tenancy module that implements `scope/2` for your data layer.

       ### Strategy C: Adopter-implemented behaviour callback

       For Stripe-Connect-style (verified payload field) or Shopify-style
       (per-shop request header) strategies:

       ```elixir
       defmodule MyApp.Tenancy do
         @behaviour Mailglass.Tenancy

         @impl true
         def scope(query, _ctx), do: ... # your scope/2 impl

         @impl true
         def resolve_webhook_tenant(%{headers: headers}) do
           case List.keyfind(headers, "x-shopify-shop-domain", 0) do
             {_, shop_domain} -> {:ok, shop_domain}
             nil -> {:error, :missing_shop_domain}
           end
         end
       end

       config :mailglass, tenancy: MyApp.Tenancy
       ```

       Returning `{:error, _}` raises `%TenancyError{type: :webhook_tenant_unresolved}`
       and returns HTTP 422.

       Postmark publishes ~13 webhook IPs at

       To enable:

       > is unset.

       ## 5. Orphan reconciliation (Oban cron)

       When a webhook arrives BEFORE the matching Delivery row commits
       (a real race for low-latency providers), mailglass inserts the
       event with `delivery_id: nil + needs_reconciliation: true`. The
       `Mailglass.Webhook.Reconciler` Oban worker sweeps these orphans
       every 5 minutes and APPENDS a `:reconciled` event when the matching
       Delivery later commits.

       Wire the cron in your Oban config:

       ```elixir
       config :my_app, Oban,
         repo: MyApp.Repo,
         plugins: [
           {Oban.Plugins.Cron, crontab: [
             {"*/5 * * * *", Mailglass.Webhook.Reconciler},
             {"0 4 * * *", Mailglass.Webhook.Pruner}      # daily prune at 4 AM
           ]}
         ],
         queues: [
           mailglass_reconcile: 1,
           mailglass_maintenance: 1
         ]
       ```

       ### If you don't use Oban

       Run the mix tasks from your own cron:

       ```bash
       # System cron / Kubernetes CronJob
       */5 * * * *  cd /app && mix mailglass.reconcile
       0 4 * * *    cd /app && mix mailglass.webhooks.prune
       ```

       mailglass emits a single boot-time `Logger.warning` if Oban is
       absent, directing you here.

       ## 6. Webhook event retention (Pruner)

       Three knobs in `Mailglass.Config`:

       ```elixir
       config :mailglass, :webhook_retention,
         succeeded_days: 14,           # default
         dead_days: 90,                # default
         failed_days: :infinity        # default — investigatable, never auto-pruned
       ```

       Set any knob to `:infinity` to disable that prune class.

       mailglass does NOT auto-prune on PII triggers. For targeted GDPR
       erasure, query directly:

       ```elixir
       from(w in Mailglass.Webhook.WebhookEvent,
         where: fragment("?->>'to' = ?", w.raw_payload, ^email)
       ) |> MyApp.Repo.delete_all()
       ```

       ## 7. Statement timeout runbook (D-29)

       mailglass sets `SET LOCAL statement_timeout = '2s'` and `SET LOCAL
       lock_timeout = '500ms'` inside the ingest transaction. This bounds
       the worst-case query latency, breaking the provider-retry-storm
       feedback loop.

       ### Symptom: sustained 5xx under load

       1. Provider retries amplify (Postmark retries 10× over 45 minutes)
       2. Retries land on a slowing DB → more DB pressure → more 5xx
       3. Loop continues until DB CPU saturates

       ### Mitigation (already in place)

       The 2s timeout means an unhealthy ingest fails fast, returns 500 to
       the provider, the provider backs off per its retry schedule, and
       the DB recovers. No log spam, no unbounded latency growth.

       ### v0.5 escape-hatch: async ingest

       If your normalize/ingest step starts taking >1s consistently
       (large adopter-extending normalizers, batched-projection workloads),
       v0.5 ships `:webhook_ingest_mode: :async` for opt-in deferred
       processing via Oban. v0.1 keeps sync default for predictable
       latency + zero ledger-loss risk.
